    <script>
        // 섹션 전환 함수
        function showSection(sectionId) {
            const sections = document.querySelectorAll('.section');
            sections.forEach(section => {
                section.classList.remove('active');
            });
            document.getElementById(sectionId).classList.add('active');
        }

        // 단어 카드 뒤집기
        function flipCard(card) {
            card.classList.toggle('flipped');
        }

        // 퀴즈 관련 변수
        let currentQuestion = 0;
        let score = 0;
        let answered = false;

        const questions = [
            {
                question: "다음 중 'beautiful'의 비교급은?",
                options: ["beautifuler", "more beautiful", "most beautiful", "beautifullest"],
                correct: 1
            },
            {
                question: "'I ___ to school yesterday.' 빈칸에 알맞은 것은?",
                options: ["go", "goes", "went", "going"],
                correct: 2
            },
            {
                question: "현재진행형으로 올바른 문장은?",
                options: ["I am study now", "I studying now", "I am studying now", "I study now"],
                correct: 2
            },
            {
                question: "'Happy'의 반대말은?",
                options: ["Sad", "Angry", "Tired", "Hungry"],
                correct: 0
            },
            {
                question: "'They ___ playing soccer now.' 빈칸에 알맞은 것은?",
                options: ["is", "are", "was", "were"],
                correct: 1
            }
        ];

        // 퀴즈 답 체크
        function checkAnswer(element, isCorrect) {
            if (answered) return;
            
            answered = true;
            const options = document.querySelectorAll('.quiz-option');
            options.forEach(option => {
                option.style.pointerEvents = 'none';
            });

            if (isCorrect) {
                element.classList.add('correct');
                score++;
            } else {
                element.classList.add('incorrect');
                // 정답 표시
                options[questions[currentQuestion].correct].classList.add('correct');
            }

            document.getElementById('next-btn').style.display = 'block';
        }

        // 다음 문제
        function nextQuestion() {
            currentQuestion++;
            answered = false;

            if (currentQuestion < questions.length) {
                const q = questions[currentQuestion];
                document.getElementById('question-number').textContent = `문제 ${currentQuestion + 1}/${questions.length}`;
                document.getElementById('question-text').textContent = q.question;
                
                const optionsHtml = q.options.map((option, index) => 
                    `<div class="quiz-option" onclick="checkAnswer(this, ${index === q.correct})">${option}</div>`
                ).join('');
                
                document.querySelector('.quiz-options').innerHTML = optionsHtml;
                document.getElementById('next-btn').style.display = 'none';
            } else {
                // 퀴즈 종료
                document.getElementById('quiz-content').innerHTML = `
                    <h3>퀴즈 완료!</h3>
                    <p style="font-size: 2em; text-align: center; margin: 30px 0;">
                        점수: ${score}/${questions.length}
                    </p>
                    <button class="btn" onclick="resetQuiz()">다시 도전하기</button>
                `;
            }
        }

        // 퀴즈 리셋
        function resetQuiz() {
            currentQuestion = 0;
            score = 0;
            answered = false;
            
            const q = questions[0];
            document.getElementById('quiz-content').innerHTML = `
                <div class="quiz-question">
                    <span id="question-number">문제 1/${questions.length}</span><br>
                    <span id="question-text">${q.question}</span>
                </div>
                <div class="quiz-options">
                    ${q.options.map((option, index) => 
                        `<div class="quiz-option" onclick="checkAnswer(this, ${index === q.correct})">${option}</div>`
                    ).join('')}
                </div>
            `;
            document.getElementById('next-btn').style.display = 'none';
        }

        // 매칭 게임 관련
        let gameCards = [];
        let flippedCards = [];
        let matchedPairs = 0;
        let gameTimer = 0;
        let timerInterval;

        const gameWords = [
            {eng: "Book", kor: "책"},
            {eng: "School", kor: "학교"},
            {eng: "Friend", kor: "친구"},
            {eng: "Happy", kor: "행복한"},
            {eng: "Study", kor: "공부하다"},
            {eng: "Music", kor: "음악"},
            {eng: "Family", kor: "가족"},
            {eng: "Love", kor: "사랑"}
        ];

        // 게임 초기화
        function initGame() {
            const gameBoard = document.getElementById('game-board');
            gameBoard.innerHTML = '';
            gameCards = [];
            flippedCards = [];
            matchedPairs = 0;
            gameTimer = 0;
            
            // 카드 생성
            gameWords.forEach((word, index) => {
                gameCards.push({id: index, text: word.eng, pair: index});
                gameCards.push({id: index + 8, text: word.kor, pair: index});
            });
            
            // 카드 섞기
            gameCards.sort(() => Math.random() - 0.5);
            
            // 카드 표시
            gameCards.forEach((card, index) => {
                const cardElement = document.createElement('div');
                cardElement.className = 'game-card';
                cardElement.textContent = '?';
                cardElement.onclick = () => flipGameCard(index);
                gameBoard.appendChild(cardElement);
            });
            
            // 타이머 시작
            clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                gameTimer++;
                document.getElementById('game-timer').textContent = `시간: ${gameTimer}초`;
            }, 1000);
        }

        // 게임 카드 뒤집기
        function flipGameCard(index) {
            if (flippedCards.length >= 2) return;
            if (flippedCards.includes(index)) return;
            
            const cards = document.querySelectorAll('.game-card');
            const card = cards[index];
            
            if (card.classList.contains('matched')) return;
            
            card.textContent = gameCards[index].text;
            card.classList.add('flipped');
            flippedCards.push(index);
            
            if (flippedCards.length === 2) {
                setTimeout(() => checkMatch(), 1000);
            }
        }

        // 매칭 확인
        function checkMatch() {
            const cards = document.querySelectorAll('.game-card');
            const [first, second] = flippedCards;
            
            if (gameCards[first].pair === gameCards[second].pair) {
                cards[first].classList.add('matched');
                cards[second].classList.add('matched');
                matchedPairs++;
                document.getElementById('matches').textContent = matchedPairs;
                
                if (matchedPairs === 8) {
                    clearInterval(timerInterval);
                    setTimeout(() => {
                        alert(`게임 완료! 소요 시간: ${gameTimer}초`);
                    }, 500);
                }
            } else {
                cards[first].textContent = '?';
                cards[second].textContent = '?';
                cards[first].classList.remove('flipped');
                cards[second].classList.remove('flipped');
            }
            
            flippedCards = [];
        }

        // 게임 리셋
        function resetGame() {
            initGame();
            document.getElementById('matches').textContent = '0';
        }

        // 페이지 로드 시 게임 초기화
        window.onload = function() {
            initGame();
        };

        // 학습 시간 추적 (실제 구현 시 localStorage 활용)
        let studyTime = 0;
        let studyInterval;

        function startStudyTimer() {
            studyInterval = setInterval(() => {
                studyTime++;
                // localStorage에 저장하여 진도 추적
                localStorage.setItem('studyTime', studyTime);
            }, 60000); // 1분마다 업데이트
        }

        // 페이지 로드 시 학습 타이머 시작
        startStudyTimer();

        // 페이지 떠날 때 타이머 정지
        window.addEventListener('beforeunload', () => {
            clearInterval(studyInterval);
        });
    </script>
